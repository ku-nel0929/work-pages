import qrcode import matplotlib.pyplot as plt # ホームページのURL url = "https://ku-nel0929.github.io/work-pages/" # 事務所のテーマカラー（ネイビー） primary_color = '#0f2350' # QRコード生成 qr = qrcode.QRCode( version=1, error_correction=qrcode.constants.ERROR_CORRECT_H, # 汚れに強くする box_size=20, # 高画質化 border=2, # 余白は最小限 ) qr.add_data(url) qr.make(fit=True) # 色を指定して画像作成 img = qr.make_image(fill_color=primary_color, back_color="white") # 画像表示 plt.figure(figsize=(5, 5)) plt.imshow(img) plt.axis('off') plt.show()
